Kemudian lihat pada App/Service/examService.php

<?php
namespace App\Service

use System\Core\Http\Request;

class examService
{
    public function register()
    {
        echo 'service examService berjalan...';
    }
}

Daftarkan pada kernel Config/Kernel.php

<?php
namespace Config;

class Kernel
{
    public $middlewareRoute = [
        'Auth' =>  \App\Middleware\Auth::class
    ];
    
    
    public $serviceRoute    = [
        'csrf' => \App\Service\CSRF_Provider::class,
        
        'exam' => \App\Service\examService::class //new service
    ];
}

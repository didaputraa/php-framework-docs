Pendaftaran pada Kernel

Buka Config/Kernel.php

<?php
namespace Config;

class Kernel
{
    public $middlewareRoute = [
        'Auth'   =>  \App\Middleware\Auth::class,
        'isAdmin'=> \App\Middleware\isAdmin::class //new middleware
    ];
}

Penerapan pada route

route::get('/')->middleware(['Auth' ,'isAdmin']);

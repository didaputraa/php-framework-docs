Auth

namespace yg digunakan

use System\Core\Auth;

public function auth(Request $input)
{
    $data = [
        'username' => $input->post('user'),
        'password' => $input->post('pass')
    ];
    
    if(Auth::attempt($data))
    {
        return redirect('dashboard/member');
    }
    else
    {
        return redirect('login');
    }
}

// cek pengguna apakah sudah login dengan cara manual
if(Auth::check())
{
    print_r(Auth::detail());
}

Auth::logout();

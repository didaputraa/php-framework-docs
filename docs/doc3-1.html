Model Fungsi Umum

Fungsi Umum pada Model

Select data

User::select(['username','password'])->get();

User::select(['username'])->addSelect('md5(password) as enkripsi')->get();

Where

User::where('username', 'dida')->get();

User::where('tall', '>=', 172)->get();

User::whereRaw("username='dida' && password=md5('dida')")->get();

Order by

User::orderBy('username', 'asc')->all();

/*
    asc | desc
*/

Group by

User::groupBy('id')->all();

Get Data

User::get() 
#limit data pada method ini adalah 0,25

User::first() 
#output yg dihasilkan adalah 1 data saja

User::all() 
#output semua data akan di tampilkan

User::count() 
#menghasilkan jumlah keseluruhan data pada tabel user
